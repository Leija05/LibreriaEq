INICIO RÁPIDO (3 PASOS)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Copia equalizer.browser.js a tu proyecto

  2. Crea tu HTML:
     <audio id="audio" controls crossorigin="anonymous">
       <source src="cancion.mp3">
     </audio>
     <script type="module" src="app.js"></script>

  3. Crea tu JavaScript (app.js):
     import { EqualizerEngine, equalizerPresets } from './equalizer.browser.js';
     
     const audio = document.getElementById('audio');
     const context = new AudioContext();
     const source = context.createMediaElementSource(audio);
     const eq = new EqualizerEngine(context);
     
     source.connect(eq.input);
     eq.connect(context.destination);
     
     // ¡Listo! Ahora controla el ecualizador
     eq.setPreset(equalizerPresets.rock);
